  5. 排序算法

    冒泡排序BubbleSort，选择排序SelectSort，插入排序InsertSort:O(n2);快速排序QuickSort:O(N*logN)

    • A
    public static void SortD(int a[],int l,int r){
     if(l>=r)
       return;
     int i = l; int j = r; int key = a[l];
     while(i<j){
         while(i<j && a[j]>=key)
             j--;
         if(i<j){
             a[i] = a[j];
             i++;
         }
         while(i<j && a[i]<key)
             i++;
         if(i<j){
             a[j] = a[i];
             j--;
         }
     }
     a[i] = key;
     SortD(a, l, i-1);
     SortD(a, i+1, r);
 }

     public static void SortB(int array[],int lenth){
   for(int i=0;i<lenth-1;i++){
       int minIndex = i;
       for(int j=i+1;j<lenth;j++){
          if(array[j]<array[minIndex]){
              minIndex = j;
          }
       }
       if(minIndex != i){
           int temp = array[i];
           array[i] = array[minIndex];
           array[minIndex] = temp;
       }
   }
 }
